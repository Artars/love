tree("Root")
	fallback
		tree("Hunt")
		tree("Wander")

//Will try to find the player
tree("Hunt")
	sequence 
		FoundEnemy
		SetPositionToEnemy
		while
			FoundEnemy
			mute
				sequence
					mute 
						fallback
							sequence
								CanEnemyBeSeen
								CloseToTarget(20.0)
								Stop
							sequence
								HasReachedDestination
								SetPositionToEnemy
					sequence
						AimAtEnemy
						CanEnemyBeSeen
						AimCorrect
						Shoot

tree("Wander")
	sequence
		not 
			FoundEnemy
		HasReachedDestination
		SetRandomPosition
