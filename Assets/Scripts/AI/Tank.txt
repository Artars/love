tree("Root")
	fallback
		tree("Hunt")
		tree("Wander")

//Will try to find the player
tree("Hunt")
	sequence 
		FoundEnemy
		Stop
		while
			FoundEnemy
			sequence
				AimAtEnemy
				CanEnemyBeSeen
				AimCorrect
				Shoot

tree("Wander")
	sequence
		not 
			FoundEnemy
		sequence
			HasReachedDestination
			GetRandomPosition
			GoToTarget
