tree("Root")
	parallel
		tree("Sense") //Sense all the time
		fallback
			tree("Hunt")
			tree("Wander")

//Will look for the player
tree("Sense")
	sequence
		Wait(100)
		Succeed

//Will try to find the player
tree("Hunt")
	sequence
		FoundEnemy()
		Say("Hunting")

tree("Wander")
	sequence
		not 
			FoundEnemy()
		sequence
			HasReachedDestination()
			GetRandomPosition()
			GoToTarget()
